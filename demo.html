💻 API使用例

// QRコード生成
import QRCode from './dist/js-qrcode.js';

const qrCode = QRCode.generate('Hello, World!', {
    errorCorrectionLevel: 'M',
    margin: 4,
    moduleSize: 4,
    color: '#000000'
});

// QRコード読み取り
const result = await QRCode.decode(imageData);
console.log(result.data);
